// Internal helper for preview
function PreviewElement({ config, tag }: { config: ElementConfig, tag: 'div' | 'span' }) {
    const style: React.CSSProperties = {
        backgroundColor: config.appearance.themeColor,
        borderRadius: `${config.appearance.borderRadius}px`,
        boxShadow: config.appearance.shadow.enabled 
            ? `0px 4px ${config.appearance.shadow.intensity}px rgba(0,0,0,0.2)` 
            : 'none',
        width: config.appearance.width === 'auto' ? (tag === 'div' ? '64px' : 'auto') : config.appearance.width,
        height: config.appearance.height === 'auto' ? (tag === 'div' ? '64px' : 'auto') : config.appearance.height,
        marginTop: `${config.appearance.margin.top}px`,
        marginRight: `${config.appearance.margin.right}px`,
        marginBottom: `${config.appearance.margin.bottom}px`,
        marginLeft: `${config.appearance.margin.left}px`,
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'center',
        // For span, if themeColor is applied as color, we remove bg
    };
 
    if (tag === 'span') {
        style.backgroundColor = 'transparent';
        style.color = config.appearance.themeColor;
    }
 
    return (
        <div 
            style={style}
            className="transition-all duration-300"
        >
             {tag === 'div' ? 'Icon' : 'Gemini Clone'}
        </div>
    );
}
